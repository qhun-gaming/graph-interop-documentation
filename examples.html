GraphInterop Help

Examples

This section provides complete examples of using GraphInterop for common scenarios.

Example 1: Math Expression Graph

Define Nodes

[NodeInfo(Category = "Math", Title = "Add")] public class AddNode : GraphNode { [Input] public float a; [Input] public float b; [Output] public float result; public override void Process(GraphContext context) { result = a + b; } } [NodeInfo(Category = "Math", Title = "Multiply")] public class MultiplyNode : GraphNode { [Input] public float a; [Input] public float b; [Output] public float result; public override void Process(GraphContext context) { result = a * b; } } [NodeInfo(Category = "Constants", Title = "Float")] public class FloatNode : GraphNode { [Input(ShowEmbedded = true, Required = false)] [SerializeField] private float _value; [Output] public float output; public override void Process(GraphContext context) { output = _value; } }

Create Graph Asset

[CreateAssetMenu(menuName = "Math Graph")] public class MathGraphAsset : GraphAsset { }

Build and Execute

// Create graph: (2 + 3) * 4 = 20 var graph = ScriptableObject.CreateInstance<MathGraphAsset>(); var const2 = new FloatNode(); const2.GetType().GetField("_value", BindingFlags.NonPublic | BindingFlags.Instance) .SetValue(const2, 2f); var const3 = new FloatNode(); const3.GetType().GetField("_value", BindingFlags.NonPublic | BindingFlags.Instance) .SetValue(const3, 3f); var const4 = new FloatNode(); const4.GetType().GetField("_value", BindingFlags.NonPublic | BindingFlags.Instance) .SetValue(const4, 4f); var add = new AddNode(); var multiply = new MultiplyNode(); graph.AddNode(const2); graph.AddNode(const3); graph.AddNode(const4); graph.AddNode(add); graph.AddNode(multiply); // Connect: const2 → add.a, const3 → add.b graph.AddConnection(const2.Id, "output", add.Id, "a"); graph.AddConnection(const3.Id, "output", add.Id, "b"); // Connect: add.result → multiply.a, const4 → multiply.b graph.AddConnection(add.Id, "result", multiply.Id, "a"); graph.AddConnection(const4.Id, "output", multiply.Id, "b"); // Execute var executor = new GraphExecutor(); executor.Execute(graph, new GraphContext()); // Result: multiply.result = 20 Debug.Log($"Result: {multiply.result}");

Example 2: Spline Processing Pipeline

Nodes

[NodeInfo(Category = "Spline/Input", Title = "Spline Source")] public class SplineInputNode : GraphNode { [Output] public ISplineContainer spline; public override void Process(GraphContext context) { spline = context.Get<ISplineContainer>("spline"); } } [NodeInfo(Category = "Spline/Process", Title = "Sample Points")] public class SamplePointsNode : GraphNode { [Input(Required = true)] public ISplineContainer spline; [Input(ShowEmbedded = true)] public float spacing = 1f; [Output] public List<Vector3> points; public override void Process(GraphContext context) { points = new List<Vector3>(); var length = spline.Spline.GetLength(); for (float t = 0; t <= length; t += spacing) { var normalized = t / length; var pos = spline.Spline.EvaluatePosition(normalized); points.Add(pos); } } } [NodeInfo(Category = "Transform", Title = "Random Offset")] public class RandomOffsetNode : GraphNode { [Input(Required = true)] public List<Vector3> points; [Input(ShowEmbedded = true)] public float maxOffset = 0.5f; [Output] public List<Vector3> result; public override void Process(GraphContext context) { result = points.Select(p => p + Random.insideUnitSphere * maxOffset).ToList(); } } [NodeInfo(Category = "Output", Title = "Instantiate Prefabs")] public class InstantiatePrefabsNode : GraphNode { [Input(Required = true)] public List<Vector3> positions; [Input(ShowEmbedded = true)] public GameObject prefab; public override void Process(GraphContext context) { foreach (var pos in positions) { Object.Instantiate(prefab, pos, Quaternion.identity); } } }

Usage

// Setup context var context = new GraphContext(); context.Set("spline", mySplineContainer); // Execute pipeline var executor = new GraphExecutor(); executor.Execute(pipelineGraph, context);

Example 3: Using Blackboard Variables

Setup

[NodeInfo(Category = "Variables", Title = "Get Variable")] public class GetVariableNode : GraphNode { public string VariableId; [Output] public object value; public override void Process(GraphContext context) { var graph = context.Get<GraphAsset>("graph"); var variable = graph.Blackboard.GetVariable(VariableId); value = variable?.Value; } }

Drag from Blackboard

In the editor, drag a variable from the blackboard to automatically create a GetVariableNode connected to that variable.

Example 4: Editor Window Integration

public class MyPipelineEditor : GraphInteropEditorWindow { [MenuItem("Window/My Pipeline Editor")] public static void Open() { var window = GetWindow<MyPipelineEditor>(); window.titleContent = new GUIContent("Pipeline Editor"); } protected override void OnEnable() { base.OnEnable(); // Auto-load last edited graph var lastPath = EditorPrefs.GetString("LastPipelineGraph", ""); if (!string.IsNullOrEmpty(lastPath)) { var asset = AssetDatabase.LoadAssetAtPath<GraphAsset>(lastPath); if (asset != null) LoadGraph(asset); } } protected override void OnGraphLoaded(GraphAsset asset) { base.OnGraphLoaded(asset); EditorPrefs.SetString("LastPipelineGraph", AssetDatabase.GetAssetPath(asset)); } protected override bool CanCreateNodeType(Type nodeType) { // Only show our custom nodes return nodeType.Namespace == "MyPipeline.Nodes"; } }

Example 5: Custom Type Converters

[InitializeOnLoadMethod] static void RegisterCustomConverters() { // Convert Color to Vector4 TypeCompatibility.RegisterConverter<Color, Vector4>( c => new Vector4(c.r, c.g, c.b, c.a)); // Convert Vector3 to float (magnitude) TypeCompatibility.RegisterConverter<Vector3, float>( v => v.magnitude); // Convert float array to AnimationCurve TypeCompatibility.RegisterConverter<float[], AnimationCurve>( values => { var curve = new AnimationCurve(); for (int i = 0; i < values.Length; i++) { curve.AddKey(i / (float)(values.Length - 1), values[i]); } return curve; }); }
