v2.0: Subgraph System

Status: ğŸ“‹ Planned (Major Version)

Description: Nested graph support

Subgraph Assets

[CreateAssetMenu(menuName = "GraphInterop/Subgraph")] public class MySubgraph : GraphAsset { // Define input variables (exposed as input ports) public override IEnumerable<ContextVariableDefinition> GetInputVariables() { yield return new ContextVariableDefinition( "input", typeof(float), isRequired: true ); } // Define output variables (exposed as output ports) public override IEnumerable<ContextVariableDefinition> GetOutputVariables() { yield return new ContextVariableDefinition( "result", typeof(float), isRequired: true ); } }

Subgraph Nodes

[NodeInfo(DisplayName = "My Subgraph")] public class MySubgraphNode : GraphNode, ISubgraphNode { [SerializeField] private GraphAsset subgraph; public GraphAsset GetSubgraph() => subgraph; protected override void OnDefinePorts(IPortDefinitionContext context) { // Ports auto-generated from subgraph's input/output variables foreach (var input in subgraph.GetInputVariables()) { context.AddInputPort(input.name) .WithDataType(input.type) .Build(); } foreach (var output in subgraph.GetOutputVariables()) { context.AddOutputPort(output.name) .WithDataType(output.type) .Build(); } } public override void Process(GraphContext context) { // Execute subgraph var subContext = new GraphContext(); // Pass inputs foreach (var input in subgraph.GetInputVariables()) { var value = this.GetInputValue<object>(input.name, context); subContext.SetContextVariable(input.name, value); } // Execute var executor = new GraphExecutor(subgraph); await executor.ExecuteAsync(subContext); // Retrieve outputs foreach (var output in subgraph.GetOutputVariables()) { var value = subContext.GetContextVariable(output.name); this.SetOutputValue(output.name, value, context); } } }

v2.1: Context and Block Nodes

Status: ğŸ“‹ Planned

Description: Hierarchical node composition

Context Nodes

public abstract class ContextNode : GraphNode { public int blockCount { get; } public BlockNode GetBlock(int index) { ... } public IEnumerable<BlockNode> blockNodes { get; } }

Example: Sequence Node

[NodeInfo(DisplayName = "Sequence")] public class SequenceNode : ContextNode { protected override void OnDefinePorts(IPortDefinitionContext context) { context.AddInputPort("execute").Build(); context.AddOutputPort("completed").Build(); } public override void Process(GraphContext context) { // Execute each block in order foreach (var block in blockNodes) { block.Process(context); } } }

Block Nodes

public abstract class BlockNode : GraphNode { public ContextNode contextNode { get; } public int index { get; } // Position in parent }

Example: Sequence Step

[NodeInfo(DisplayName = "Step")] [UseWithContext(typeof(SequenceNode))] public class SequenceStepNode : BlockNode { protected override void OnDefinePorts(IPortDefinitionContext context) { context.AddInputPort<string>("action").Build(); } public override void Process(GraphContext context) { var action = this.GetInputValue<string>("action", context); Debug.Log($"Step {index}: {action}"); } }

Use Cases

  • State Machines: ContextNode = state machine, BlockNodes = states

  • Sequences: Ordered execution steps

  • Branches: If/Else blocks

  • Loops: For/While bodies

v3.0: GraphToolkit UI Backend

Status: ğŸ”® Future (Major Version)

Description: Use GraphToolkit's UI instead of GraphView

Why Wait?

  • GraphToolkit UI is still experimental (0.4.0-exp.2)

  • Missing features (minimap, copy/paste, etc.)

  • API stability concerns

When Ready:

[InitializeOnLoadMethod] private static void RegisterGTKBackend() { GraphInteropEditorWindow.RegisterBackend(() => new GraphToolkitEditorBackend()); } // Users can switch backends via preference GraphInteropSettings.PreferredBackend = "GraphToolkit";

Benefits:

  • Official Unity support

  • Advanced edge routing

  • Better performance (potentially)

  • Future Unity features

No User Code Changes:

  • Backend abstraction handles everything

  • Same runtime API

  • Same graph assets
